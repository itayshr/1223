import tkinter as tk import webbrowser from tkinter import simpledialog, messagebox # מילון שמכיל את שמות המשתמשים ומערכות השעות שלהם users_schedule = {} # רשימת הימים לפי הסדר days_order = ["ראשון", "שני", "שלישי", "רביעי", "חמישי"] available_subjects = ["ערבית", "מדעים", "מתמטיקה", "תרבות ישראל", "בישול ואפייה", "חינוך", "ספורט", "עברית", "ניהול פיננסי", "אנגלית", "ממרם"] # פונקציה להכנסת מערכת שעות למשתמש חדש def create_new_user(username): # יצירת מערכת שעות ריקה (המשתמש יזין את המקצועות שלו) new_schedule = {} for day in days_order: # יצירת חלון לכל יום עם כפתורים לבחירת מקצועות selected_subjects = [] def toggle_subject(subject): if subject in selected_subjects: selected_subjects.remove(subject) else: selected_subjects.append(subject) # יצירת חלון חלופי לבחור מקצועות day_window = tk.Toplevel(root) day_window.title(f"בחירת מקצועות ליום {day}") label = tk.Label(day_window, text=f"בחר את המקצועות ליום {day}:") label.pack() for subject in available_subjects: button = tk.Checkbutton(day_window, text=subject, command=lambda s=subject: toggle_subject(s)) button.pack() def save_day(): new_schedule[day] = set(selected_subjects) day_window.destroy() save_button = tk.Button(day_window, text="שמור", command=save_day) save_button.pack() day_window.mainloop() # הוספת המשתמש החדש למילון users_schedule[username] = new_schedule messagebox.showinfo("הרשמה", f"✅ נרשמת בהצלחה, {username}! מערכת השעות שלך מוכנה.") # פונקציה להתחברות עם משתמש קיים def login_user(username): if username in users_schedule: return True else: return False # חלון ראשי root = tk.Tk() root.title("מערכת שעות") # פונקציה לפתיחת אתר בגוגל def open_github(): try: # פתיחת האתר בגיטאהאב webbrowser.open("https://github.com/itayshr/1223") print("הדפדפן נפתח בהצלחה") except Exception as e: print(f"שגיאה בפתיחת הדפדפן: {e}") messagebox.showerror("שגיאה", "⚠ לא ניתן לפתוח את הדפדפן. בדוק את החיבור לאינטרנט.") # בקשה מהמשתמש לבחור אם להירשם או להתחבר def register_or_login(): action = messagebox.askquestion("בחר פעולה", "האם אתה רוצה להירשם (כן) או להתחבר (לא)?") if action == "yes": # בקשת שם משתמש חדש username = simpledialog.askstring("שם משתמש", "אנא הזן שם משתמש חדש:") if username in users_schedule: messagebox.showwarning("שגיאה", "⚠ שם המשתמש הזה כבר קיים.") else: create_new_user(username) elif action == "no": # בקשת שם משתמש קיים username = simpledialog.askstring("שם משתמש", "אנא הזן את שם המשתמש שלך:") if login_user(username): messagebox.showinfo("ברוך הבא", f"\nשלום {username}, ברוך הבא למערכת!") else: messagebox.showwarning("שגיאה", "⚠ שם המשתמש לא קיים במערכת.") # כפתור לפתיחת אתר ב-GitHub github_button = tk.Button(root, text="פתח את GitHub", command=open_github) github_button.pack(pady=20) # קריאה לפונקציה כדי לבחור בין רישום או התחברות register_or_login() # לאחר שהמשתמש נכנס או נרשם, אפשר להמשיך בשאר הקוד root.mainloop()
